#### 最长子串



```
#include <stdlib.h>
#include <iostream>
#include <string>

#ifndef NULL
#define NULL
#endif

int** initDp(int n, int m) {
    int** dp = new int*[n];
    for (int i = 0; i < n; ++i) {
        dp[i] = new int[m];
    }
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            dp[i][j] = 0;
        }
    }
    return dp;
}

int** getDp(const char* str1, int len1, const char* str2, int len2) {
    if (str1 == NULL || str2 == NULL) {
        return NULL;
    }
    int** dp = initDp(len1, len2);
    for (int i = 0; i < len1; ++i) {
        if (str1[i] == str2[0]) {
            dp[i][0] = 1;
        } else {
            dp[i][0] = 0;
        }
    }
    for (int j = 1; j < len2; ++j) {
        if (str1[0] == str2[j]) {
            dp[0][j] = 1;
        } else {
            dp[0][j] = 0;
        }
    }
    for (int i = 1; i < len1; ++i) {
        for (int j = 1; j < len2; ++j) {
            if (str1[i] == str2[j]) {
                dp[i][j] = dp[i - 1][j - 1] + 1;
            }
        }
    }
    return dp;
}

int main() {
    std::string str1 = "dp3456xj";
    std::string str2 = "k3456jj";
    int** dp = getDp(str1.c_str(), str1.length(), str2.c_str(), str2.length());
    int end = 0;
    int max = 0;
    for (int i = 0; i < str1.length(); ++i) {
        for (int j = 0; j < str2.length(); ++j) {
            if (dp[i][j] > max) {
                max = dp[i][j];
                end = i;
            }
        }
    }
    std::cout << "max : " << max << std::endl;
    std::cout << str1.substr(end - max + 1, max) << std::endl;
    return 0;
}
```
