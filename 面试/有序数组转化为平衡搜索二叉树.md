#### 有序数组转化为平衡搜索二叉树



```
/**
 * 有序数组转化为平衡二叉树
 */
#include <stdlib.h>
#include <iostream>
#include <vector>

struct Node {
    Node* left;
    Node* right;
    int val;
    Node(int v) {
        left = right = NULL;
        val = v;
    }
};

Node* arrayToTree(std::vector<int>& arr, int start, int end) {
    if (start > end) {
        return NULL;
    }
    int mid = start + (end - start) / 2;
    Node* root = new Node(arr[mid]);
    root->left = arrayToTree(arr, start, mid - 1);
    root->right = arrayToTree(arr, mid + 1, end);
    return root;
}

void preShow(Node* root) {
    if (root == NULL) {
        return;
    }
    preShow(root->left);
    std::cout << root->val << ", ";
    preShow(root->right);
}

int main() {
    std::vector<int> arr{1, 2, 3, 4, 5, 6, 7, 8};
    Node* root = arrayToTree(arr, 0, arr.size() - 1);
    preShow(root);
    std::cout << std::endl;
    return 0;
}
```
