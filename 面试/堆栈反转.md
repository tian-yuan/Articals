#### 堆栈反转



```
#include <stdlib.h>
#include <iostream>
#include <stack>

int getAndRemoveLastEle(std::stack<int>& mystack) {
    int result = mystack.top();
    mystack.pop();
    if (mystack.empty()) {
        return result;
    } else {
        int i = getAndRemoveLastEle(mystack);
        mystack.push(result);
        return i;
    }
}

void reverse(std::stack<int>& mystack) {
    if (mystack.empty()) {
        return;
    }
    int i = getAndRemoveLastEle(mystack);
    reverse(mystack);
    mystack.push(i);
}

void show(std::stack<int>& mystack) {
    if (mystack.empty()) {
        return;
    }
    int i = mystack.top();
    std::cout << i << ", ";
    mystack.pop();
    show(mystack);
    mystack.push(i);
}

int main() {
    std::stack<int> mystack;
    for (int i = 0; i < 10; ++i) {
        mystack.push(i);
    }
    show(mystack);
    std::cout << std::endl;
    show(mystack);
    std::cout << std::endl;
    reverse(mystack);
    show(mystack);
    std::cout << std::endl;

    return 0;
}
```
