#### 冒泡排序



```
#include <stdio.h>
#include <iostream>
#include <vector>

void show(std::vector<int>& input) {
    for (int i = 0; i < input.size(); ++i) {
        std::cout << input[i] << ", ";
    }
    std::cout << std::endl;
}

void sort(std::vector<int>& input) {
    if (input.size() <= 0) {
        return;
    }
    int endIndex = input.size() - 1;
    int max = 0;
    int maxIndex = 0;
    while (endIndex > 0) {
        max = input[0];
        maxIndex = 0;
        for (int j = 1; j <= endIndex; ++j) {
            if (input[j] > max) {
                max = input[j];
                maxIndex = j;
            }
        }
        int temp = input[endIndex];
        input[endIndex] = max;
        input[maxIndex] = temp;
        endIndex--;
        show(input);
    }
}

int main() {
    //std::vector<int> input{5, 7, 1, 2, 9, 4, 5};
    std::vector<int> input{1, -1};
    show(input);
    sort(input);
    show(input);
    return 0;
}
```
