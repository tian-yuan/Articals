#### 判断两个字符串是否为旋转字符串



```
/*
判断两个字符串是否为旋转字符串
*/
#include <iostream>
#include <string>
#include <stdlib.h>

int main() {
    // std::cout << "begin." << std::endl;
    std::string a = "cdab";
    std::string b = "abcd";
    std::string con = a + a;
    std::cout << "con : " << con << std::endl;
    std::string::size_type pos = con.find(b);
    std::cout << "pos : " << pos << std::endl;
    std::cout << "npos : " << std::string::npos << std::endl;
    if (pos == std::string::npos) {
        std::cout << "b : " << b << " is not rotate string of a : " << a << std::endl;
    } else {
        std::cout << "b : " << b << " is rotate string of a : " << a << std::endl;
    }
    return 0;
}
```
