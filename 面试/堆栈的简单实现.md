#### 堆栈的简单实现



```
struct Node {
    Node* pre;
    Node* next;
    int val;
    Node(int x) {
        val = x;
        pre = next = NULL;
    }
};

struct LinkList {
    Node* head;
    Node* tail;

    LinkList() {
        head = tail = NULL;
    }

    void push(int val) {
        Node* node = new Node(val);
        push(node);
    }

    void push(Node* node) {
        if (head == NULL) {
            head = tail = node;
            tail->pre = node;
        } else {
            Node* tmp = tail;
            tail->next = node;
            tail = node;
            tail->pre = tmp;
        }
    }

    Node* pop() {
        if (head == NULL) {
            return NULL;
        }
        if (head == tail) {
            Node* tmp = head;
            head = tail = NULL;
            return tmp;
        } else {
            Node* tmp = tail;
            tail->pre->next = NULL;
            tail = tail->pre;
            return tmp;
        }
    }
};
```
